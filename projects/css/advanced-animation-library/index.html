3D Cube Rotation:

@keyframes cube-rotate {
  0% { transform: rotateX(0deg) rotateY(0deg) rotateZ(0deg); }
  25% { transform: rotateX(90deg) rotateY(0deg) rotateZ(0deg); }
  50% { transform: rotateX(90deg) rotateY(90deg) rotateZ(0deg); }
  75% { transform: rotateX(90deg) rotateY(90deg) rotateZ(90deg); }
  100% { transform: rotateX(0deg) rotateY(0deg) rotateZ(0deg); }
}

.cube-3d {
  transform-style: preserve-3d;
  perspective: 1000px;
}

.cube-3d.animate {
  animation: cube-rotate 3s infinite linear;
}

Morphing Shapes:

@keyframes morph-cycle {
  0% { border-radius: 50%; transform: scale(1) rotate(0deg); }
  25% { border-radius: 0%; transform: scale(1.2) rotate(90deg); }
  50% { border-radius: 50% 0% 50% 0%; transform: scale(0.8) rotate(180deg); }
  75% { border-radius: 0% 50% 0% 50%; transform: scale(1.1) rotate(270deg); }
  100% { border-radius: 50%; transform: scale(1) rotate(360deg); }
}

.morph-shape.animate {
  animation: morph-cycle 2s infinite ease-in-out;
}

Glitch Effect:

@keyframes glitch-animation {
  0% { transform: translate(0); }
  20% { transform: translate(-2px, 2px); }
  40% { transform: translate(-2px, -2px); }
  60% { transform: translate(2px, 2px); }
  80% { transform: translate(2px, -2px); }
  100% { transform: translate(0); }
}

.glitch-text.animate {
  animation: glitch-animation 0.3s infinite;
  position: relative;
}

.glitch-text.animate::before,
.glitch-text.animate::after {
  content: attr(data-text);
  position: absolute;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
}

.glitch-text.animate::before {
  animation: glitch-animation 0.3s infinite;
  color: #ff0000;
  z-index: -1;
}

.glitch-text.animate::after {
  animation: glitch-animation 0.3s infinite reverse;
  color: #00ff00;
  z-index: -2;
}
